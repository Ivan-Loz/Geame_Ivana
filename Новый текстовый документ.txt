private void button1_Click_1(object sender, EventArgs e)
{
    string un = textBox1.Text.Trim();
    string pd = textBox2.Text;

    if (stream == null  !client.Connected)
    {
        ShowError(labelError, "Нет подключения к серверу.");
        return;
    }

    if (string.IsNullOrEmpty(un)  string.IsNullOrEmpty(pd))
    {
        ShowError(labelError, "Введите логин и пароль.");
        return;
    }

    try
    {
        byte[] cmd = Encoding.UTF8.GetBytes("login\n");
        stream.Write(cmd, 0, cmd.Length);

        byte[] data = Encoding.UTF8.GetBytes(un + "\n");
        stream.Write(data, 0, data.Length);

        byte[] data1 = Encoding.UTF8.GetBytes(pd + "\n");
        stream.Write(data1, 0, data1.Length);

        byte[] buffer = new byte[256];
        int bytesRead = stream.Read(buffer, 0, buffer.Length);

        if (bytesRead == 0)
        {
            ShowError(labelError, "Сервер закрыл соединение.");
            return;
        }

        bool rb = bool.Parse(Encoding.UTF8.GetString(buffer, 0, bytesRead));

        if (rb)
        {
            Form2 form2 = new Form2();
            form2.Show();
            this.Hide();
        }
        else
        {
            ShowError(labelError, "Неверный логин или пароль.");
        }
    }
    catch (Exception ex)
    {
        ShowError(labelError, ex.Message);
    }
}
